Câu lệnh WHERE
    được sử dụng để filter ( lọc ) dữ liệu từ bảnh
    ví dụ 1:
    SELECT * FROM customers WHERE id=5
    ( khi WHERE thì nếu cột là số thì không cần dấu nháy )
    ví dụ 2:
    SELECT * FROM customers WHERE company ='Company A'
    lấy ra tất cả các bản ghi mà cột company = 'Company A'
    Nếu WHERE cột là chuỗi thì phải chuổi trong dấy nháy '' hay ""
